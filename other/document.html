document对象的属性如下：

cookie属性:
有效值（会话期间，单位秒，）、作用域（path属性，domain属性）
读取：document.cookie
    // 返回一个对象形式的cookie
    //存储cookie时 用encodeURIComponent()函数进行编码.
    function getCookies() {
        var cookies = {};           // 定义一个对象
        var all = document.cookie;  // 获取cookie； e.g：AreaCode=330300; UserName=guest; IsRememberUser=1; username=guest; password=123456;
        if (all === "")             // 判断cookie是否为空
         {
               return cookies;//是的话 返回空的cookie
         }
        var list = all.split(";"); // 按“；”划分 cookie； e.g：AreaCode=330300", " UserName=guest", " IsRememberUser=1", " username=guest", " password=123456"
        for(var i = 0; i < list.length; i++) {  // 遍历cookie
            var cookie = list[i];
            var p = cookie.indexOf("=");        // Find the first = sign
            var name = cookie.substring(0,p);   // Get cookie name
            var value = cookie.substring(p+1);  // Get cookie value
            value = decodeURIComponent(value);  // cookie值 解码
            cookies[name] = value;              // 将名/值存到对象里
        }
        return cookies;//返回对象格式的cookie
    }
getElementById属性：（id）

getElementsByName属性：（name属性）
var x=document.getElementsByName("myInput");
getElementsByTagName属性：(标签)
